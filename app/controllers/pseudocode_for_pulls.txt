#matching indorm pulls with the dorm they said they would pull into
if student.round == in_dorm
    if student.dormchoice != pull.dorm 
        don't allow pull
    else
        allow pull
    end
end

#whitelisting and blacklisting for Anonymized Pre-Placements (pg. 6)
if Anonymized_Pre-Placement in pull.suite
    if A_P-P.whitelist == true && student not in A_P-P.whitelist
        don't allow pull
    elseif A_P-P.whitelist == true && student in A_P-P.blacklist
        don't allow pull
    else
        allow pull
    end
end
